Write a program that computes the commission and total payment of sales of an employee. This program will:
  • Ask for sale item amount.
  • Repeat until data is done being entered.
  • Calculate the percentage of commission based on total sales.
Go back to main page.

    function init()
        {
            var sale;
            var saleValue;
            var salestotal;
            var total;
            var salestotalValue;
            var commission = 0.10;
            var x;

            x = 1;
            total = 0;
            sale = 0;
            saleValue = 0;
            salestotal = 0;

            while (sale != "q" && sale != "Q") {
                salestotal +=  saleValue;
                sale = window.prompt("Enter a sale amount (Type 'Q' to end ) :  ", "0");
                saleValue = parseFloat(sale);
                if (sale != "q" && sale != "Q") {
                    document.write("Sale " + x + ": $" + saleValue + "
");
                    x++;
                }
            }
            salestotalValue = parseFloat(salestotal);
            total = salestotalValue * commission;
            document.write("
Total Commission : " + "$" + total + "");
        }
