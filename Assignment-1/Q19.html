Program 19

Write a program that calculates whether a customer has exceeded the credit limit of an account. To do this:
  • Obtain account number, balance previous, total charges, total payments/credits, and credit limit.
  • Calculate current debt balance
  • Compare that the debt is < credit limit, if so print leftover credit, if not print credit exceeded
Go back to main page.

Result

Source Code

        function init() {
            var account;
            var balance = 0;
            var totalCharge = 0;
            var totalCredit = 0;
            var creditLimit = 0;
            var newBalance = 0;
            var leftCredit = 0;

            account = window.prompt("Account Number: ");
            creditLimit = parseInt(window.prompt("Credit Limit : "));
            balance = parseInt(window.prompt("Balance at beginning of month: "));
            totalCharge = parseInt(window.prompt("Total charges this month: "));
            totalCredit = parseInt(window.prompt("Total amount of payments this month: "));

            newBalance = balance + totalCharge - totalCredit;
           
            if (newBalance > creditLimit)
            {
               document.write("Credit Limit exceeded");
            }
            else
            {
                leftCredit = creditLimit - newBalance;
                document.write("Left over Credit : " + leftCredit);
            }
        }
