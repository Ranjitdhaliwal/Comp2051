Program 18

Write a program that calculates an average of miles per gallon for a vehicle over the history of miles/gallons factor. The steps are the following:
  • Input number of full tanks to average.
  • Input the gallons of fuel filled for number x
  • Input the miles driven for number x
  • Repeat entry of gallons and miles until the number of tanks is entered.
  • Print the average miles per gallon for each tankful and print an average at the bottom.
Go back to main page.

Result

Source Code

function init() {
    var totalTanks = 0;
    var miles = 0;
    var totalMiles = 0;
    var gallons = 0;
    var totalGallons = 0;
    var milesPerGallon =0 ;
    var totalMilesPerGallon =0;
 
    count = 1;
    totalTanks = window.prompt("Enter number of full tanks to average:");
           
    while (count <= totalTanks)
    {
            // Get gallon of fuel filled for current tank[count]
            gallons = parseInt(window.prompt("Enter gallon of fuel filled for tank #" + count + ":"));
                    
                // Get miles driven for current tank[count]
            miles = parseInt(window.prompt("Enter miles driven for tank #" + count + ":"));
                   
            totalMiles += miles;
            totalGallons += gallons;
                    
            if ( gallons != 0 )
            {
            milesPerGallon = miles / gallons;
            document.writeln(
                    "<br />Gallons used: " + gallons +
                        "<br />Miles driven: " + miles +
                        "<br />Miles/Gallon : " + milesPerGallon );
                    } // end if
                  
            count++;
    } // end while

    totalMilesPerGallon = totalMiles / totalGallons;
    document.writeln("<br /><br />Total Miles/Gallon: " + totalMilesPerGallon);

}
