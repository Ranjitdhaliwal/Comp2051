A class of ten students took a quiz. The grades (integers in the range 0 to 100) for this quiz are available to you. Determine the class average on the quiz.

function init() {
            var count = 1;
            var totalGrades = 0;
            var average = 0;
            while (count <= 10)
            {
                totalGrades += parseInt(prompt("Enter grade for student number #" + count+"\n[ Grade must be between 0 and 100]","0"));
                count++;
            }
            average = totalGrades / 10;
            alert("Class average is : " + average);
            }
