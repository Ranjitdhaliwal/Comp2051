Write a script that calculates the product of the odd integers from 1 to 15 then outputs HTML text that displays the results.

Go back to main page.

Source Code

 
function init() {
            var count = 1;
            var product = 1;
          
            while (count <= 15)
            {
                if (count % 2 != 0)
                {
                    product *= count; // Multiply if it is a odd number
                }

                count++;
            }
           
            alert("Product of odd number between 1 - 15 (including 15) is : " + product);
        }
