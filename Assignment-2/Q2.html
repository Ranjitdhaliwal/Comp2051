Develop a class-averaging program that will process an arbitrary number of grades each time the program is run.

function init() {
            var count = 1;
            var totalStudent = parseInt(prompt("Enter number of student", "0"))
            if (totalStudent > 0) {
                var totalGrades = 0;
                var average = 0;
                while (count <= totalStudent) {
                    totalGrades += parseInt(prompt("Enter grade for student number #" + count + "\n[ Grade must be between 0 and 100]", "0"));
                    count++;
                }
                average = totalGrades / totalStudent;
                alert("Class average is : " + average);
            }
            else
                alert("Number of students must be greater then 0.")
            }
